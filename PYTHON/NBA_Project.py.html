# NBA Data Cleaning & Feature Engineering
import pandas as pd

# ---- Paths ----
base_path = r"C:\Users\neofo\Downloads\archive"

# ---- Load CSVs ----
players = pd.read_csv(f"{base_path}\\Players.csv", low_memory=False)
player_stats = pd.read_csv(f"{base_path}\\PlayerStatistics.csv", low_memory=False)
games = pd.read_csv(f"{base_path}\\Games.csv", low_memory=False)

# ---- Convert date columns ----
player_stats['gameDate'] = pd.to_datetime(player_stats['gameDate'], errors='coerce')
games['gameDate'] = pd.to_datetime(games['gameDate'], errors='coerce')

# ---- Fill missing values ----
player_stats.fillna(0, inplace=True)
games.fillna(0, inplace=True)

# ---- Feature Engineering ----
# 1. Efficiency: (points + rebounds + assists) / minutes
player_stats['efficiency'] = (player_stats['points'] +
                              player_stats['reboundsTotal'] +
                              player_stats['assists']) / player_stats['numMinutes'].replace(0,1)

# 2. Points per Minute
player_stats['points_per_min'] = player_stats['points'] / player_stats['numMinutes'].replace(0,1)

# 3. Rebound Ratio
player_stats['rebound_ratio'] = player_stats['reboundsTotal'] / player_stats['numMinutes'].replace(0,1)

# ---- Export cleaned & feature-enhanced CSVs ----
player_stats.to_csv(f"{base_path}\\cleaned_PlayerStatistics.csv", index=False)
players.to_csv(f"{base_path}\\cleaned_Players.csv", index=False)
games.to_csv(f"{base_path}\\cleaned_Games.csv", index=False)

print("âœ… CSVs cleaned, features added, and exported for BigQuery.")
